    }
    return result
}

func FindMinimum(numbers []int) int {
    if len(numbers) == 0 {
        return 0
    }
    min := numbers[0]
    for _, num := range numbers {
        if num < min {
            min = num
        }
    }
    return min
}

func FindMapDifferences(map1, map2 map[string]int) ([]string, []string, []string) {
    var only1, only2, different []string
    
    // Проверяем ключи из первой мапы
    for key, value1 := range map1 {
        if value2, exists := map2[key]; exists {
            if value1 != value2 {
                different = append(different, key)
            }
        } else {
            only1 = append(only1, key)
        }
    }

    for key := range map2 {
        if _, exists := map1[key]; !exists {
            only2 = append(only2, key)
        }
    }
    
    return only1, only2, different
}

Что такое слайс? - глубокое понимание

Слайс - это динамический массив в Go, один из самых мощных и часто используемых типов данных. Слайс является ссылочным типом и состоит из трех компонентов:

  • pointer - указатель на underlying array (подлежащий массив)
  • length - текущее количество элементов (len)
  • capacity - максимальное количество элементов без реаллокации (cap)
// Визуализация слайса
slice := []int{1, 2, 3, 4, 5}
fmt.Printf("len=%d cap=%d %v\n", len(slice), cap(slice), slice)
// Вывод: len=5 cap=5 [1 2 3 4 5]

func insertAt(slice []int, index int, value int)[]int {
    // Увеличиваем размер слайса на 1
    slice = append(slice, 0)
    // Сдвигаем элементы вправо
    copy(slice[index+1:], slice[index:])

    slice[index] = value
    return slice

func insertSliceAt(slice []int, index int, values []int) []int {
    // Создаем место для новых элементов
    slice = append(slice, make([]int,
    len(values))...)
    // Сдвигаем существующие элементы
    copy(slice[index+len(values):],
    slice[index:])
    // Копируем новые элементы
    copy(slice[index:], values)
    return slice
}

}

// Универсальная функция преобразования
func transform(numbers []int, fn func(int) int) []int {
    result := make([]int, len(numbers))
    for i, num := range numbers {
        result[i] = fn(num)
    }
    return result
}

// Использование
numbers := []int{1, 2, 3, 4, 5}
doubled := doubleAll(numbers) // [2, 4, 6, 8, 10]
squared := transform(numbers, func(n int) int {
    return n * n
}) // [1, 4, 9, 16, 25]

// Проблема: слайс держит ссылку на большой массив

    return data[len(data)-10:] // может удерживать весь data в памяти!
}

// Решение: копирование нужных данных
func getLastElementsSafe(data []int) []int {
    lastElements := data[len(data)-10:]
    result := make([]int, len(lastElements))
    copy(result, lastElements)
    return result // data может быть освобожден сборщиком мусора
}

Что такое мапа? - глубокое понимание

Мапа (map) - это встроенный ассоциативный тип данных в Go, который хранит пары ключ-значение. Мапы реализованы как хеш-таблицы и обеспечивают быстрый доступ к данным.

🔑 Ключевые особенности мап

  • Неупорядоченная коллекция (порядок элементов не гарантируется)
  • Быстрый поиск: O(1) в среднем случае
  • Ключи должны быть сравнимыми (comparable) типами
  • Значения могут быть любого типа
  • Ссылочный тип (reference type)
  • Нулевое значение: nil
// Базовый синтаксис
var m map[KeyType]ValueType

// Пример
ages := map[string]int{
    "Алиса": 25,
    "Боб": 30,
    "Карл": 35,
}
fmt.Printf("Количество элементов: %d\n", len(ages))

func getValue(matrix map[string]map[string]int, row, col string) (int, bool) {
    if rowMap, rowExists := matrix[row]; rowExists {
        if value, colExists := rowMap[col]; colExists {
            return value, true
        }
    }
    return 0, false
}

        matrix[row] = make(map[string]int)
    }
    matrix[row][col] = value
}

✅ Рекомендации

🎯 Когда использовать мапы

  • Быстрый поиск элементов по ключу
  • Подсчет частоты элементов
  • Группировка данных
  • Кэширование результатов вычислений
  • Реализация множеств (set) - map[T]bool
  • Индексирование данных

🚫 Когда НЕ использовать мапы

  • Когда важен порядок элементов - используйте слайсы
  • Для небольших коллекций (<10 элементов) - слайс может быть быстрее
  • Когда нужна сортировка - используйте слайсы + sort
  • Для конкурентного доступа - используйте sync.Map или мьютексы
